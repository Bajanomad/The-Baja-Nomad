const feeds = [ { name: "The Cabo Sun", url: "https://api.rss2json.com/v1/api.json?rss_url=https%3A%2F%2Fthecabosun.com%2Ffeed" }, { name: "Gringo Gazette", url: "https://api.rss2json.com/v1/api.json?rss_url=https%3A%2F%2Fwww.gringogazette.com%2Ffeed" }, { name: "La Paz Times", url: "https://api.rss2json.com/v1/api.json?rss_url=https%3A%2F%2Flapaztimes.com%2Ffeed" } ]; const container = document.getElementById('news-container'); // Pre-create feed divs in order so they always appear in the right slots const feedDivs = feeds.map(feed => { const div = document.createElement('div'); div.className = 'feed'; div.innerHTML = `

${feed.name}

Loading...

`; container.appendChild(div); return div; }); function formatDate(dateStr) { const date = new Date(dateStr); return date.toLocaleDateString(undefined, { year: 'numeric', month: 'short', day: 'numeric' }); } feeds.forEach((feed, index) => { fetch(feed.url) .then(response => response.json()) .then(data => { const div = feedDivs[index]; if (!data || !data.items || !data.items.length) { div.innerHTML = `

${feed.name}

No articles found or feed blocked.

`; return; } const list = document.createElement('ul'); list.className = 'articles'; data.items.slice(0, 8).forEach(item => { const li = document.createElement('li'); const a = document.createElement('a'); a.href = item.link; a.target = '_blank'; a.rel = 'noopener noreferrer'; a.textContent = item.title; li.appendChild(a); const dateSpan = document.createElement('div'); dateSpan.className = 'pubDate'; dateSpan.textContent = formatDate(item.pubDate); li.appendChild(dateSpan); list.appendChild(li); }); div.innerHTML = `

${feed.name}

`; div.appendChild(list); }) .catch(() => { feedDivs[index].innerHTML = `

${feed.name}

Failed to load feed.

`; }); });
